import java.io.*;

public class Main {
    public static void main(String[] args) throws Exception {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        // Remove spaces
        String S = br.readLine().replaceAll(" ", "").trim();
        int L = Integer.parseInt(br.readLine().trim());

        int n = S.length();

        if (L > n || L == 0) {
            System.out.println(0);
            return;
        }

        int maxA = 0;
        int currentA = 0;

        for (int i = 0; i < L; i++) {
            if (S.charAt(i) == 'a') {
                currentA++;
            }
        }

        maxA = currentA;

        for (int i = L; i < n; i++) {

            if (S.charAt(i - L) == 'a') {
                currentA--;
            }

            if (S.charAt(i) == 'a') {
                currentA++;
            }

            maxA = Math.max(maxA, currentA);
        }

        System.out.println(maxA);
    }
}
