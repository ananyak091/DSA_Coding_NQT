import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws Exception {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));

        String startDay = br.readLine().trim().toLowerCase();
        int n = Integer.parseInt(br.readLine().trim());

        // Map days
        Map<String, Integer> map = new HashMap<>();
        map.put("sun", 0);
        map.put("mon", 1);
        map.put("tue", 2);
        map.put("wed", 3);
        map.put("thu", 4);
        map.put("fri", 5);
        map.put("sat", 6);

        int startIndex = map.get(startDay);

        int distance = (7 - startIndex) % 7;

        int count = 0;

        if (distance < n) {
            count = 1 + (n - distance - 1) / 7;
        }

        System.out.println(count);
    }
}
