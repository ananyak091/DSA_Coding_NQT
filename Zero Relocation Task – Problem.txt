You are given:

An integer N

An array of N integers

Your task is to:

Move all zeros in the array to the end, while maintaining the relative order of non-zero elements.

You must do it in-place (without using extra array ideally).

Input:
6
0 1 0 3 12 0

Output:
1 3 12 0 0 0 *}


import java.io.*;
import java.util.*;

public class Main {
    public static void main(String[] args) throws Exception {
        
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        
        
        int n = Integer.parseInt(br.readLine().trim());
        String[] input = br.readLine().trim().split("\\s+");
        
        int[] arr = new int[n];
        for(int i = 0; i < n; i++){
            arr[i] = Integer.parseInt(input[i]);
        }
        
        int index =0;
        
        for(int i=0; i<n; i++) {
            // Move non-zero elements in front of array
            if(arr[i] != 0) {
                arr[index] = arr[i];
                index++;
            }
        }
        
        // Copy back the '0' in remaining place
        while(index < n) {
            arr[index] = 0;
            index++;
        }
        
        for(int i = 0; i < n; i++){
            System.out.print(arr[i] + " ");
        }
        
    }
}
